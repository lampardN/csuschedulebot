def deploy(groups: list, start_row: int, sheet: openpyxl.Workbook, db: sqlite3.Connection, cursor: sqlite3.Cursor, week: list[str], time: list[str]) -> None:
    """_summary_

    Args:
        groups (list): _description_
        start_row (int): _description_
        sheet (openpyxl.Workbook): _description_
        db (sqlite3.Connection): _description_
        cursor (sqlite3.Cursor): _description_
        week (list[str]): _description_
        time (list[str]): _description_
    """
    for group in groups:
        begin = start_row
        for i in range(working_days := 6):
            for j in range(number_of_obj := 7):
                obj = sheet.cell(row=begin, column=group['id'] + 1).value
                aud = sheet.cell(row=begin, column=group['id'] + 2).value
                educ = sheet.cell(row=begin, column=group['id'] + 3).value
                if educ:
                    # print(f'<{week[i]} -- {time[j]} -- {obj} -- {aud} -- {group["name"]} -- {educ}>')
                    cursor.execute(
                        "select * from Timetable where Date=? and Time=? and Object=? and Audience=? and \"Group\"=? and Educator=?",
                        (week[i], time[j], obj, aud, group['name'], educ))
                    if not cursor.fetchall():
                        cursor.execute("insert into Timetable values (?,?,?,?,?,?)",
                                       (week[i], time[j], obj, aud, group['name'], educ))

                begin += 1
    db.commit()


async def parse(message: types.Message, file_name: str) -> None:
    """_summary_

    Args:
        message (types.Message): _description_
        file_name (str): _description_
    """
    try:
        wb: openpyxl.Workbook = openpyxl.load_workbook(filename=file_name, data_only=True)
        sheet: openpyxl.Workbook = wb[wb.sheetnames[0]]

        try:
            db: sqlite3.Connection = sqlite3.connect("CSU_Data.db")
            cursor: sqlite3.Cursor = db.cursor()
        except:
            await message.answer("Не могу выполнить ваш запрос, база открыта другим пользователем")
            return

        i: int = 0
        start_row: int = 0
        groups: list[dict] = []
        for row in sheet.values:
            i += 1
            if isinstance(row[0], str):
                if row[0].lower().strip() == 'дни':
                    groups: list[dict] = [{'name': row[i], 'id': i} for i in range(3, len(row)) if row[i] is not None]
                    start_row: int = i + 2
                    break

        time: list[str] = ['9-00', '10-45', '13-00', '14-45', '16-25', '18-15', '19-45']
        week: list[str] = []
        date_row: int = start_row
        for i in range(working_days:=6) :
            dates = sheet[f'A{date_row}':f'A{date_row + working_days}']
            cur_date: datetime
            for cell in dates:
                if isinstance(cell[0].value, datetime):
                    cur_date = cell[0].value
                    break
            if not cur_date:
                continue
            cur_date: date = cur_date.date()
            cur_date: str = cur_date.strftime('%d.%m.%y')
            week.append(cur_date)
            date_row += 7
        #print(working_days)
        deploy(groups, start_row, sheet, db, cursor, week, time)

        db.close()
        await message.answer("Данные успешно добавлены")
        await message.delete()
    except:
        await message.answer("Не удалось обработать документ")


async def getFile(message: types.Message) -> None:
    """_summary_

    Args:
        message (types.Message): _description_
    """
    if DBController().getEntryByID(message.chat.id)["Type"] == "Admins":
        try:
            await message.answer("Начинаю загрузку документа")
            save_dir = "./"
            file_name = message.document.file_name
            await message.document.download(destination_file=save_dir+file_name)
            await message.answer("Документ загружен, начинаю обработку")
            await parse(message, file_name)
        except:
            await message.answer("Произошла непредвиденная ошибка")
    else:
        await message.delete()
        await message.answer('У вас недостаточно прав')
